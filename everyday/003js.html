# instanceof 和typeof

  • typeof只适合查看值类型，基本分不清引用类型的具体类型
  • instanceof
f instanceof Foo//f的__proto__一层一层往上，看能否对应到Foo.prototype
var a=new Array();
alert(a instanceof Array); // true
var a=new test();
alert(a instanceof test) //true
  • Object.prototype.toString.call()
Object.prototype.toString.call('') ;   // [object String]
Object.prototype.toString.call(1) ;    // [object Number]

不使用自身的toString()原因,这是因为toString为Object的原型方法，而Array 、Function等类型作为Object的实例，都重写了toString方法。

# 阻止冒泡

在微软的模型中，你必须设置事件的cancelBubble的属性为true

window.event.cancelBubble = true

在w3c模型中你必须调用事件的stopPropagation()方法

e.stopPropagation()

跨浏览器解决方案：

function doSomething(e) {
 if (!e) {
 var e = window.event;
 e.cancelBubble = true;
 }

 if (e.stopPropagation) {
 e.stopPropagation();
 }
 
}

# 兼容ie监听

var x = document.getElementById("myBtn");
if (x.addEventListener) {                    //所有主流浏览器，除了 IE 8 及更早 IE版本
    x.addEventListener("click", myFunction);
} else if (x.attachEvent) {                  // IE 8 及更早 IE 版本
    x.attachEvent("onclick", myFunction);
}

function bindEvent(elem,type,selector,fn){
	 if(fn==null){
	    fn =selector;
	    selector=null; 
	}
	if(elem.addEventListener){
		 elem.addEventListener(type,function(e){
			 console.log('addevent')
		   appleuse(e)
		})
	}else if(elem.attachEvent){
		elem.attachEvent("on"+type,function(){
			var e = window.event;
			console.log('attach');
			console.log(e);
			appleuse(e);
		})
	}
	function appleuse(e){
		console.log(e)
		if(selector){
				console.log(e.srcElement)
		    selector=selector.toUpperCase();
			var tem=e.target||e.srcElement;
		    if(tem.nodeName===selector){
		        fn(e)
		    }
		}else{
		    fn(e)
		}
	}
}
  • ie低版本下无法直接通过函数直接获取e，需要window.event去获取
  • ie低版本下无target，如果需要获取点击元素的nodeName需要获取另一个属性srcElement来获取
  • e.target 指向触发事件监听的对象,e.currentTarget 指向添加监听事件的对象.

# async/await

function getSyncTime() {
  return new Promise((resolve, reject) => {
    try {
      let startTime = new Date().getTime()
      setTimeout(() => {
        let endTime = new Date().getTime()
        let data = endTime - startTime
        resolve( data )
      }, 500)
    } catch ( err ) {
      reject( err )
    }
  })
}

async function getSyncData() {
  let time = await getSyncTime()
  let data = `endTime - startTime = ${time}`
  return data
}

async function getData() {
  let data = await getSyncData()
  console.log( data )
}

getData()
  • 可以让异步逻辑用同步写法实现
  • 最底层的await返回需要是Promise对象
  • 可以通过多层 async function 的同步写法代替传统的callback嵌套
