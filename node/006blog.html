# 接口开发

# http请求概述

  1. DNS解析，建立TCP连接 ，发送http请求
  2. server接受到http请求，处理后返回
  3. 客户端接受到返回数据，处理数据(页面渲染，执行js等)

# nodejs处理http请求

get请求

 const http = require('http')
 const querystring=require('querystring')
 const server=http.createServer((req,res)=>{
 	console.log(req.method)
 	const url=req.url;
 	console.log(url)
	//http://localhost:8000/today/go?name=12&ag=19
 	req.message=querystring.parse(url.split('?')[1]) ;
 	res.end(JSON.stringify(req.message))
	//{"name":"12","ag":"19"}
 })
 server.listen(8000,()=>{
 	console.log('this server is work')
 })

post请求

安装postman(谷歌插件)

 const http = require('http')
 const querystring=require('querystring')
 const server=http.createServer((req,res)=>{
 	if(req.method==='POST'){
 		console.log(req.headers['content-type'])
 		let postData="";
 		req.on('data',thunk=>{
 			postData+=thunk.toString()
 		})
 		req.on('end',()=>{
			console.log(postData)
 			res.end(postData)
 		})
 	}
 	
 })
 server.listen(8000,()=>{
 	console.log('this server is work')
 })
 /*{
	"name":"zhangsan",
	"sex":"man",
	"age":40
}*/

 const http = require('http')
 const querystring=require('querystring')
 const server=http.createServer((req,res)=>{
 	const{url,method}=req;
 	const path=url.split("?")[0];
 	const query1=url.split("?")[1];
 	const query2=querystring.parse(url.split("?")[1]);
 	res.setHeader('Content-type','application/json')
 	const obj={
 		url,
 		method,
 		query1,
 		query2,
 		path
 	}
 	if(method=='GET'){
 		res.end(JSON.stringify(obj))
		//http://localhost:8000/?name=wei&age=20
		/*{
			"url":"/?name=wei&age=20",
			"method":"GET",
			"query1":"name=wei&age=20",
			"query2":{"name":"wei","age":"20"},
			"path":"/"
			}*/
 	}else if(method==='POST'){
 		console.log(obj)
 		res.end(JSON.stringify(obj))
		//curl -d 'name=lili&age=20' http://localhost:8000
		/*{ url: '/',
		  method: 'POST',
		  query1: undefined,
		  query2: [Object: null prototype] {},
		  path: '/' }*/
 	}
 	
 	
 })
 server.listen(8000,()=>{
 	console.log('this server is work')
 })
