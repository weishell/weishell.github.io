# 响应式原理

<!DOCTYPE html>
<html>
	<head>
		<meta charset="utf-8">
		<title>响应式</title>
	</head>
	<body>
		<div id="app"></div>
	</body>
</html>
<script type="text/javascript">
	let obj={}
	Object.defineProperty(obj,"name",{
		get(){
			return document.querySelector("#app").innerHTML
		},
		set(val){
			document.querySelector("#app").innerHTML =val
		}
	})
	
	obj.name="wang"
</script>
