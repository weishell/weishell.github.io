# vuex状态管理模式

每一个 Vuex 应用的核心就是 store（仓库）。“store”基本上就是一个容器，它包含着你的应用中大部分的状态 (state)。 当页面刷新的时候，vuex的数据会丢失(重置)，所以，有的数据刷新可以丢失，有的不可以，如果不可以的话，需要配合localstorage来存储。

# 安装vuex

npm install vuex --save

理解

vuex理解 官方解释：vuex是一个专为vue.js应用程序开发的 状态管理模式。它采用集中式存储管理应用的所有的状态，并以相应的规则保证状态以一种可预测的方式发生变化。 可以理解为是一个 全局变量

  • state 数据中心
  • mutations 操作数据
  • actions 什么时候触发操作，执行动作，改变数据

# mapAction

使用 mapActions 辅助函数将组件的 methods 映射为 store.dispatch 调用（需要先在根节点注入 store）

import {mapActions} from 'vuex'

export default {
  name: 'Home',
  methods:{
	 
	  // ...mapActions(['miss']),
	  ...mapActions({add:'addtwice',miss:'miss'})
	  
  }
  
}

# mapMutations

使用 mapMutations 辅助函数将组件中的 methods 映射为 store.commit 调用

import {mapMutations} from 'vuex'

export default {
  
  methods:{

	  // ...mapMutations({add:'add',miss:'miss'})
	  ...mapMutations(['miss','add']),
  }
  
}

// 1.维护状态state
// 2.修改状态commit
// 3.业务逻辑控制dispatch
// 4.状态派发getter
// 5.实现state响应式
// 6.插件
// 7.混入

let Vue;

function install(_Vue) {
  Vue = _Vue;

  // 混入：把store选项指定到Vue原型上
  Vue.mixin({
    beforeCreate() {
      if (this.$options.store) {
        Vue.prototype.$store = this.$options.store;
      }
    }
  });
}

class Store {
  // options:  {state:{count:0}, mutations: {count(state){}}}
  constructor(options = {}) {
	  console.log(options)
    // 利用vue数据响应式
    this.state = new Vue({
      data: options.state
    });

    // 初始化mutaions
    this.mutations = options.mutations || {};
    this.actions = options.actions || {};

    options.getters && this.handleGetters(options.getters)
  }

  // 触发mutaions，需要实现commit
  commit = (type, arg) => {
	  console.log(type)
    // this指向Store实例
    const fn = this.mutations[type]; //获取变更函数
    fn(this.state, arg);
  };

  dispatch = (type, arg) => {
    const fn = this.actions[type];
    return fn({ commit: this.commit, state: this.state }, arg);
  }

  // {getters: {score(state){return state.xx}}}
  handleGetters(getters) {
    this.getters = {}; // store实例上的getters

    // 定义只读的属性
    Object.keys(getters).forEach(key => {
        Object.defineProperty(this.getters, key, {
            get: () => {
                return getters[key](this.state);
            }
        })
    })
  }
}
// Vuex
export default { Store, install };

import Vue from "vue";
import KVuex from "../kvuex";

Vue.use(KVuex, '$s');

export default new KVuex.Store({
  state: {
    count: 0
  },
  mutations: {
    add(state, num = 1) {
      state.count += num;
    }
  },
  getters: {
    score(state) {
      return "score:" + state.count;
    }
  },
  actions: {
    // 复杂业务逻辑，类似于controller
    // 比如ajax请求
    asyncAdd({ commit }) {
      return new Promise(resolve => {
        setTimeout(() => {
          commit("add");
          resolve({ok:1})
        }, 1000);
      })
      
    }
  }
});
