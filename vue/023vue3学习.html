# setup ref reactive

  • setup 函数是一个新的组件选项。作为在组件内使用 Composition API 的入口点,setup中和在其他地方创建的数据或者方法需要return后才能绑定和操作。
  • ref和reactive都是用来创建响应式对象的,ref是一个个，而reactive则是建立一个对象
<template>
  <div class="home">
   home
	<div>{{count}}</div>
	<button @click="add">+</button>
	<button @click="plus(count)">-</button>
	<div>money:{{state.yuan}}</div>
	<button @click="double">double</button>
	
  </div>
</template>

<script>

import {ref,reactive} from "vue"
import plus from "./plus.js"
export default {
  name: 'Home',
 
  setup(){
	   let count =ref(7)
	   function add(){
		   console.log(this.count)
		   console.log(count)
		   count.value++;//count单独使用它的value值才是真正需要的结果
		   this.count++;//加了this的话，就可以直接修改结果而不需要.value,原因？？？？
	   }
	   let state=reactive({
		   yuan:1
	   })
	   //每次单独添加太麻烦，可以使用reactive一次性添加，淡然引用的时候需要加上state了
	   function double(){
		   console.log(this.state.yuan)
		   state.yuan *=2
	   }
	 //需要使用的方法和数据需要return出去
	  return{
		  count,
		  plus,
		  add,
		  double,
		  state
	  }
  }

}
</script>

# toRefs

上面reactive的导出和使用，必须state.key,如果想像vue2中的data一样定义直接使用，可以使用 toRefs

<template>
  <div class="home">
   home
	
	<div>money:{{yuan}}</div>
	<button @click="double">double</button>

	
  </div>
</template>

<script>

import {ref,reactive,toRefs} from "vue"
export default {
  name: 'Home',
 
  setup(){
	   let state=reactive({
		   yuan:1
	   })
	   function double(){
		   console.log(this)
		   console.log(this.state.yuan===this.yuan)//true
		  this.yuan *=2
	   }
	  return{
		  double,
		  state,
		 ...toRefs(state)
	  }
  }

}
</script>

出现两个yuan，一个在this上，一个在this.state上，当然，如果注释掉return里的state，那么只有this.yuan;而this.state则是undefined

# vue3watch computed watchEffect

<template>
  <div class="home">
   home
	<div>{{num1}}</div>
	<div>{{num2}}</div>
	<button @click="increment">+</button>
	<button @click="stopwatch">stop</button>
  </div>
</template>

<script>

import { reactive, watch, computed, toRefs, watchEffect } from "vue";
export default {
  name: 'Home',
 
 setup() {
     const state = reactive({
       num1: 1,
       //定义一个计算属性
       num2: computed(() => state.num1 * 2)
     });
     
     // 如果响应性的属性有变更，就会触发这个函数,但他是惰性的
     watchEffect(() => {
       console.log(`effect 触发了！${state.num1}`);
     });
 
     // 定义一个监听器
     const stop = watch(state, (val, oldVal) => {
		 console.log(val,oldVal)//两个值相等？？？？
       console.log("watch ", oldVal.num1);
     });
 
     //数值增加方法
     const increment = () => state.num1++;
 
     // 停止监听
     const stopwatch = () => stop();
 
     return {
       ...toRefs(state),
       stopwatch,
       increment
     };
   }

}
</script>

# ref 操作dom

<template>
  <div>
    <div ref="refdemo">domref</div>
    <input type="range" v-model="size" />
  </div>
</template>
<script>
import { ref, watch, onMounted } from "vue";
export default {
  setup() {
    // 定义空白ref作为Template ref操作 dom
    const refdemo = ref(null);
	console.log(refdemo)
    const size = ref(24);
    // 监听拖动 
    watch(size, (val, oldVal) => {
      refdemo.value.style.fontSize = size.value + "px";
    });
    return {
      refdemo,
      size
    };
  }
};
</script>

# vue3 父子组件通信

和vue2的通信大同小异，新建ComDemo.vue,setup函数接受一个 props 和 context 上下文

<template>
  <div>
   <ComDemo :name="'我是父组件传值'" @talk="talk" ref="comdemo"></ComDemo>
   <button @click="action">action</button>
  </div>
</template>
<script>
import ComDemo from "./Zi";
import { ref,  onMounted,provide } from "vue";
export default {
	components: {
	  ComDemo
	},
	setup() {
	  // 定义一个ref响应式对象
	  const comdemo = ref(null);
	  onMounted(() => {
		//得到子组件的值
		console.log(comdemo.value.str);
	  });
	  function action(){
		  // 触发子组件事件
		  comdemo.value.talk();
	  }
	  function talk(val){
		  console.log(val,"子组件发来的消息")
	  }
	  return {
		comdemo,
		action,
		talk,
		
	  };
	}
};
</script>






<template>
  <div>{{name}}</div>

import { inject } from "vue";
export default {
  props: {
    name: String
  },

    // setup中使用 prop

  }

# Composition API

按功能划分，而不是之前的按生命周期划分

<template>
  <div>{{status?`${x}-${y}`:`${y}-${x}`}}</div>
</template>

<script>
import { reactive, onMounted, onUnmounted, toRefs, ref } from "vue";
// 返回鼠标位置
const useMouse = () => {
  const state = reactive({
    x: 0,
    y: 0
  });
  const update = e => {
    state.x = e.pageX;
    state.y = e.pageY;
  };
  onMounted(() => {
    window.addEventListener("mousemove", update);
  });
  onUnmounted(() => {
    window.removeEventListener("mousemove", update);
  });
  return toRefs(state);
};
// 监控键盘事件
const useKeyBoard = () => {
  const status = ref(false);
  const update = () => {
    status.value = !status.value;
  };
  onMounted(() => {
    window.addEventListener("keypress", update);
  });
  onUnmounted(() => {
    window.removeEventListener("onkeydown", update);
  });
  return {
    status
  };
};
export default {
  name: "HelloWorld",
  setup() {
    return {
      ...useMouse(),
      ...useKeyBoard()
    };
  }
};
</script>
