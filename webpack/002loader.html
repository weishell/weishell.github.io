# loader

webpack不能识别非js文件，需要借助loader打包

loader可以单独写在rules里，也可以用use包裹，如果是使用到多个loader，use应该为数组的形式

cnpm i file-loader -D

打包jpg文件

const path = require('path') 
module.exports={
	mode:'production',
	entry:'./index.js',
	module:{
		rules:[{
			test:/\.jpg$/,
			use:{
				loader:'file-loader'
			}
		}]
	},
	output:{
		filename:'bundle.js',
		path:path.resolve(__dirname,'zip')
	}
}

# url-loader

url-loader功能包含file-loader的基本功能，同时还提供一个limit来限制大小，如果文件小于limit则会直接以base64打包到js里，如果大于则会生成新文件。

cnpm i url-loader --save-dev

const path = require('path') 
module.exports={
	mode:'production',
	entry:'./index.js',
	module:{
		rules:[{
			test:/\.(jpg|gif|png)$/,
			use:{
				loader:'url-loader',
				options:{
					name:'[name].[ext]',
					outputPath:'img/',
					limit:2048
				}
			}
		}]
	},
	output:{
		filename:'bundle.js',
		path:path.resolve(__dirname,'zip')
	}
}

# css-loader

查询参数 importLoaders，用于配置「css-loader 作用于 @import 的资源之前」有多少个 loader。

module.exports = {
  module: {
    rules: [
      {
        test: /\.css$/,
        use: [
          'style-loader',
          {
            loader: 'css-loader',
            options: {
			  modules:true,//开启css模块化打包
              importLoaders: 2, 
			  // 0 => no loaders (default); 1 => postcss-loader; 2 => postcss-loader, sass-loader
            },
          },
          'postcss-loader',
          'sass-loader',
        ],
      },
    ],
  },
};

modules:true进行css模块化打包;可以避免css全局污染; import style from'./index.scss'

import apple from './apple.jpg'
import style from'./index.scss'
import demo from './demo.js'

imgs.classList.add(style.zzz);

# 编写一个loader

1一个同步loader

const loaderUtils = require('loader-utils');

module.exports = function(source) {
	return source.replace('lee', 'world');
}

2一个异步loader

注意异步loader需要引入async()

const loaderUtils = require('loader-utils');

module.exports = function(source) {
	const options = loaderUtils.getOptions(this);
	const callback = this.async();

	setTimeout(() => {
		const result = source.replace('dell', options.name);
		callback(null, result);
	}, 1000);
}

# resolveLoader配置简化自定义loader配置路径

const path = require('path');

module.exports = {
	mode: 'development',
	entry: {
		main: './src/index.js'
	},
	resolveLoader: {
		modules: ['node_modules', './loaders']
	},
	module: {
		rules: [{
			test: /\.js/,
			use: [
				{
					loader: 'replaceLoader',
				},
				{
					loader: 'replaceLoaderAsync',
					options: {
						name: 'lee'
					}
				},
			]
		}]
	},
	output: {
		path: path.resolve(__dirname, 'dist'),
		filename: '[name].js'
	}
}
