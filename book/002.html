# 设定获取属性attr

给p元素加个id

d3.select("p").attr("id","xxx")

如果要获取则只需要写出第一个参数即可

d3.select("circle").attr("cx")

(如果不能用attr获取的属性，可以用property去获得,如input的value等)

d3.select("xxx").property("value")

d3.select("xxx").property("value","xxx")

# each()

each(允许对选择集的各元素分别处理)

let persons=[{id:1,name:"apple"},{id:2,name:"purple"}];
let p=d3.selectAll("p")
p.data(persons).each((d,i)=>{
	d.age=20
}).text(d=>{
	return d.id+":"+d.age
}).exit().remove()
//1:20
//2:20

# 求值

extent返回最小和最大值，数组

let nums=[30,10,50,40,20]
let min=d3.min(nums)
let max=d3.max(nums)
let extent=d3.extent(nums)
let chuli=d3.min(nums,(d)=>{
	return d*3
})
console.log(min,max,extent,chuli)
//10 50 [10,50] 30
  • sum 和
  • mean 平均值
  • median 中间值 无返回null
let nums=[30,10,50,40,20,42]
let sum=d3.sum(nums)
let mean=d3.mean(nums,(d)=>{
	return d*2
})
let median=d3.median(nums,(d)=>{
	return d-2
})
console.log(sum,mean,median)
//192 64 33
  • quantile 分位数
let num=[1,3,10]
num.sort(d3.ascending)
console.log(d3.quantile(num,0.25))
console.log(d3.quantile(num,0.8))
//2
//7.2

# 标准差和方差

  • d3.variance(arr[,accessor]) 方差
  • d3.deviation(arr[,accessor]) 标准差 方差和标准差用于度量随机变量和均值之间的偏离程度
let arr=[1,3,5,7]
console.log(d3.variance(arr))
console.log(d3.deviation(arr))
//6.666666666666667
//2.581988897471611

# 获取位置

  • d3.bisectLeft() 获取某数组左边的位置
  • d3.bisect()/d3.bisectRight() 获取某数组右边的位置
let arr=[10,13,19,22,25,16]
console.log(arr)
let ileft=d3.bisectLeft(arr,16)
console.log(ileft)
// 2
arr.splice(ileft,0,77)
console.log(arr)
//[10, 13, 77, 19, 22, 25, 16]

16虽然在末尾，但是如果升序排列，它是第三个数，而它的左边就是2

# d3操作数组

  • d3.shuffle(arr[,lo[,hi]]) 随机排列数组
let arr=[10,13,19,22,25,16]
console.log(d3.shuffle(arr))
  • d3.merge(arrays) 合并两/多个数组
let arr=[10,13,19,22,25,16]
let arr1=[77,88]
let arr2=[96]
console.log(d3.merge([arr,arr1,arr2]))
//[10, 13, 19, 22, 25, 16, 77, 88, 96]
  • d3.pairs(arr) 返回邻接的数组对
let arr=[10,13,19]
console.log(d3.pairs(arr))
//[[10,13],[13,19]]
  • d3.range([start,]stop[,step]) 返回等差数列
let arr=[1,2,3,4,5,6,7,8]
let a=d3.range(8)
console.log(a)
//[0, 1, 2, 3, 4, 5, 6, 7]
let b=d3.range(2,8,3);
console.log(b)
// [2, 5]
  • d3.permute(arr,indexs) 根据索引号数组返回排列后的数组，原数组不变 ,超出序列补undefined
let arr=[1,2,3,4,5,6,7,8]
let newarr =d3.permute(arr,[2,1,0,50])
console.log(newarr)
// [3, 2, 1, undefined]
  • d3.zip() 用数组的第i项组成了新数组的第i项
let arr=[1,2,3,4,5,6,7,8]
let arr1 =[3,5]
let arrall=d3.zip(arr,arr1)
console.log(arrall)
//[[1,3,[2,5]]]

let arr=[1,2,3,4,5,6,7,8]
let arr1 =[3,5]
let x=d3.sum(d3.zip(arr,arr1),(d)=>{
	return d[0]*d[1]
})
console.log(x)
//13
  • d3.transpose(matrix)
let arr=[[1,2,3],[4,5,9],[4,6,7]]
let t =d3.transpose(arr)
console.log(t)
//[[1,4,4],[2,5,6],[3,9,7]]

# d3的映射

  • d3.map([object][,key]) 构造映射 第一个参数源数组 第二个为映射的key
  • map.has(key) 判断存在返回true false
  • map.get(key) key存在返回value不存在返回undefined
  • map.set(key,value)
  • map.remove(key)key存在删除key和value返回true，不存在返回false
  • map.keys() 以数组形式返回改map所有的key
  • map.values() 以数组形式返回改map所有的value
  • map.entries() 以数组形式返回map所有的key和value
  • map.forEach(function) v3版本有，现在v5版本应该废除了
  • map.empty() 映射为空为true反之false
  • map.size() 返回映射大小
let dataset=[{id:1000,color:"red"},
			 {id:1001,color:"green"},
			 {id:1002,color:"blue"}
			];
let map = d3.map(dataset,function(d){
	return d.id
})

console.log(map.has(1007))
//false

console.log(map.get(1001))
//{id:1001,color:"green"}

console.log(map.remove(1001))
//true
console.log(map.get(1001))
//undefined
console.log(map.keys())
// ["1000", "1002"]
console.log(map.values())
//[{id:1000,color:"red"},
//{id:1002,color:"blue"}]

console.log(map.entries())
//[{key:"1000",value:{id:1000,color:"red"}},
//	{key:"1002",value:{id:1002,color:"blue"}}]		

console.log(map.empty())
//false
console.log(map.size())
//2

# 集合Set

集合表示具有特定性质的食物总体

  • d3.ser([array]) 如果数组中有重复元素，则只添加一次
  • set.has(value)
  • set.add(value) 没有value添加，并返回该元素，有就不添加
  • set.remove(value)
  • set.values()
  • set.forEach() v3版本有，现在v5版本应该废除了
  • set.empty()
  • set.size()
let dataset=[1,2,3,4];
let set = d3.set(dataset)
console.log(set.size())
//4

# 嵌套结构Nest

  • d3.nest() 创建一个嵌套结构
  • nest.key(function) 指定key
  • nest.entries(array)指定数组array将被用于构建嵌套结构
let persons = [
	{id:100,name:"apple",year:1992,hometown:"北京"},
	{id:101,name:"orange",year:1993,hometown:"上海"},
	{id:102,name:"water",year:1993,hometown:"北京"},
	{id:103,name:"ok",year:1992,hometown:"广东"},
	{id:104,name:"change",year:1991,hometown:"北京"},
]

let nest=d3.nest()
		   .key(function(d){return d.year})
		   .key(function(d){return d.hometown})
		   .entries(persons)
		   
console.log(JSON.stringify(nest))

  • nest,rollup(function) 对每一组叶子节点调用指定的函数function,该函数含有一个参数values,是当前叶子节点的数组.
let persons = [
	{id:100,name:"apple",year:1992,hometown:"北京"},
	{id:101,name:"orange",year:1993,hometown:"上海"},
	{id:102,name:"water",year:1993,hometown:"北京"},
	{id:103,name:"ok",year:1992,hometown:"广东"},
	{id:104,name:"change",year:1991,hometown:"北京"},
]

let nest=d3.nest()
		   .key(function(d){return d.hometown})
		   .sortValues(function(a,b){
			   return d3.ascending(a.year,b.year)
		   }).rollup(function(val){return val.length})
		   .entries(persons)
		   
console.log(JSON.stringify(nest))
//[{"key":"北京","value":3},{"key":"上海","value":1},{"key":"广东","value":1}]
