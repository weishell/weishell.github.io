# 数组和字符串易混淆的几个方法

# 数组方法sort

sort() 方法用于对数组的元素进行排序，并返回数组。默认排序顺序是根据字符串UniCode码。返回排序后的数组

let arr=[1,111,2,20,3,19,28]
console.log(arr.sort())
//[1, 111, 19, 2, 20, 28, 3]

console.log(arr.sort((a,b)=>{return a-b}))
//[1, 2, 3, 19, 20, 28, 111]
//需要比较后得出正确的顺序

# d3中的比较数组scan方法

它返回的是位置，根据指定的比较器返回最小元素的索引，如果没有指定 comparator 则默认为 ascending(升序)

 //<script type = "text/javascript" src = "https://d3js.org/d3.v4.min.js"></script>
var array = [{foo: 42}, {foo: 91},{foo:107},{foo:19},{foo:20}];
var arr=[42,91,107,19,20]
console.log(d3.scan(arr))//3
console.log(d3.scan(array, function(a, b) { return a.foo - b.foo; }))//3
console.log(d3.scan(array, function(a, b) { return b.foo - a.foo; }))//2

# 数组方法splice

向/从数组中添加/删除项，然后返回 被删除的项 。会改变原始数组。

let arr = [1,3,5,7]
arr.splice(1,2,9,10)
console.log(arr)
//[1, 9, 10, 7]

let arr = [1,3,5,7]
arr.splice(1,0,9,10)
console.log(arr)
//[1, 9, 10, 3, 5, 7]

添加的数是加在指定位置前的

# join和split转化数组字符串

 let a='21k123k123324ksdafk3'
 let arr =a.split("k")
 console.log(arr)
 //["21", "123", "123324", "sdaf", "3"]
 let b=arr.join("--")
 console.log(b)
 //21--123--123324--sdaf--3

# indexOf数组和字符串

let arr=[1,2,3,4,5,6]
let str="123456"
console.log(arr.indexOf(3))//2
console.log(str.indexOf(45))//3

# includes 数组和字符串

let arr=[1,2,3,4,5,6]
let str="123456"
console.log(arr.includes(3))//true
console.log(str.includes(45))//true
console.log(arr.includes(2,1))//true
console.log(arr.includes(2,3))//false

# slice 数组和字符串

var arr=[1,2,3,4,5,6]
var arr2 =arr.slice(2,4)
console.log(arr)//[1,2,3,4,5,6]
console.log(arr2)//[3,4]

var str ="djqfwq";
str.slice(1,4)//jqf
