function fun(array){
	arr.forEach((el)=>{
		if(el.parentId==array.deptId){
			if(array.children){
				array.children.includes(el)?null:array.children.push(el)
			}else{
				array.children=[el]
			}
			//  array.children?array.children.push(el):array.children=[el];
			array.children.forEach((el,i)=>{
				fun(array.children[i])
			})
		}
	})
}
