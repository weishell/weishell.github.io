# react响应式设计思想和事件绑定

 import React,{Component,Fragment} from "react"
 //Fragment =>占位符
 class App extends Component{
 	constructor(props) {
 	  super(props)
 		this.state={
 			inputvalue:"",
 			list:["react"]
 		}
 	}
 	render(){
 		return(
 		<Fragment>
 			<div>
 					<input value={this.state.inputvalue} onChange={this.inputclick.bind(this)}/>
 					<button onClick={this.btnclick.bind(this)}>提交</button>
 			</div>
 			<ul>
 				{this.state.list.map((el,i)=>{
 					return <li key={i} onClick={this.deleteli.bind(this,i)}>{el}</li>
 				})}
 			</ul>
 		</Fragment>
 		)	
		//onChange=>在react中事件要驼峰写
 	}
 	inputclick(e){
 		console.log(e.target.value)
 		this.setState({
 			inputvalue:e.target.value
 		})
		//react不可以直接修改state状态的数据，需要使用setState来修改
		//相对于vue而言，对数据的变动限制更大
 		// this.state.inputvalue=e.target.value
 	}
 	btnclick(e){
 		this.setState({
 			 //list:this.state.list.push(e.target.value)
 				list:[...this.state.list,this.state.inputvalue],
 				inputvalue:""
 		})
 	}
 	deleteli(i){
 		let temarr=[...this.state.list];
		//不要直接更改state（vuex）
 		temarr.splice(i,1)
 		// alert(this.state.list)
 		this.setState({
 			list:temarr
 		})
 	}
 }
 export default App

# 组件的拆分和传值

//父组件
import React,{Component,Fragment} from "react"
import ToDisplay from "./ToDisplay"
import './style.css'

class App extends Component{
	constructor(props) {
	  super(props)
		this.state={
			inputvalue:"",
			list:["react"]
		}
	}
	//当父组件的render函数被重新运行的时候，子组件的render函数也会重新运行
	//当组件的state或者props发生变化是，rneder函数会被重新执行
	render(){
		return(
		{/*JSX=>creatElement=》虚拟dom=》真实dom*/}
		<Fragment>
			<div>
					<label htmlFor="doinput">label</label>
					<input 
					id="doinput"
					className="input"
					value={this.state.inputvalue}
					onChange={this.inputclick.bind(this)}
					/>
					<button onClick={this.btnclick.bind(this)}>提交</button>
			</div>
			<ul>
				{this.state.list.map((el,i)=>{
					return <ToDisplay 
									content={el} 
									index={i} 
									deleteli={this.deleteli.bind(this)}
									/>
				})}
			</ul>
		</Fragment>
		)	
	}
	inputclick(e){
		console.log(e.target.value)
		this.setState({
			inputvalue:e.target.value
		})
	}
	btnclick(e){
		this.setState({
				list:[...this.state.list,this.state.inputvalue],
				inputvalue:""
		})
	}
	deleteli(i){
		let temarr=[...this.state.list];
		temarr.splice(i,1)
		// alert(this.state.list)
		this.setState({
			list:temarr
		})
	}
}
export default App

//子组件
import React from 'react'
class ToDisplay extends React.Component{
	render(){
		return(
			<li  onClick={this.deleteitem.bind(this)}>{this.props.content}</li>
		)
	}
	deleteitem(){
		this.props.deleteli(this.props.index)
	}
}
export default ToDisplay

总结

  • 父子组件，父组件以属性的形式传值给子组件，包括父组件的方法都能传递
  • 父组件的方法传给子组件，要注意改变this的指向
  • 子组件接受父组件的方法或者属性，要用props来接受
